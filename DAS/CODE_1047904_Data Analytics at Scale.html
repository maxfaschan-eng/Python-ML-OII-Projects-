def sampleImages(seed=2,n_groups=3,n_images=5):
    '''
    Computes a disproportionate stratified random sample of images
  
    Parameters: 
    arg1 (int): seed setting for random group & image draw for reproducability (default value: 2)
    arg2 (int): number of strata from which to draw n_images respectively (default value: 2)
    arg3 (int): number of images to randomly sample from each strata (default value: 5)
   
    Returns: 
    df: disproportionate stratified random sample of images
    '''
    # Fetch all image strings in directory
    Image_df = pd.DataFrame(glob.glob("das_images/*.jpg"),columns=["Image_Name"])
    
    #assemble dataframe 
    Image_df["Image_Name"] = Image_df["Image_Name"].map(lambda x: x.replace('das_images/',''))
    Image_df["Image_Group"] = Image_df["Image_Name"].map(lambda x: x.replace(x[4:],''))

    #adjust column sorting 
    Image_df = Image_df.reindex(["Image_Group","Image_Name"],axis=1)
    
    #set seed for reproducability
    np.random.seed(seed)
    
    #create grouped instance
    Image_df_grouped = Image_df.groupby("Image_Group")
    
    #derive array containing all groups 
    image_groups_array=np.arange(Image_df_grouped.ngroups)
    
    #randomly shuffle array 
    np.random.shuffle(image_groups_array)
    
    #subsett original df throgh element-wise membership check against subset of randomly shuffled array
    Image_df = Image_df.loc[Image_df_grouped.ngroup().isin(image_groups_array[:n_groups ])]
    
    #randomly sample n Images per group
    Image_sub_df = Image_df.groupby('Image_Group').apply(lambda x: x.sample(n_images,replace=False,random_state=seed)).reset_index(drop=True)
    
    return Image_sub_df

######### Performance Evaluation per Hasher ############
    
def hashError(hasher,name,sample,th=0.16,hash_size=8,res_print=False,FINd=False):
    '''
    Computes many-to-many hamming distances and quantifies performance 
    stats from derived confusion matrix  
  
    Parameters: 
    arg1 (func): hasher algorithm to calculate hash keys
    arg2 (name): name of the hasher algorithm used as output identifier 
    arg3 (df): stratified random Image sample (col1: Image_Group, col2: Image_Name)
    arg4 (float): precentage threshold below which two hash keys are considered similar
    arg5 (int): bitsize of hash key (default value:8)
    arg6 (bool):print out at function run time (default value:False)
    arg7 (bool):indication whether algorithm is FINd or not (default value:False)

    Returns: 
    df: Image Groups, absolute & normalized hamming distances, actual similiarity, 
    perceived similarity, confusion Matrix categorization
  
    '''
    
    ## calculate hash keys
    temp_df = sample.copy()
    temp_df["Hash"]=np.nan
    
    if FINd == True: 

        for index_row,image in enumerate(temp_df.Image_Name):
            temp_df.loc[index_row,"Hash"] = hasher.fromFile(f"das_images/{image}")
    
    if FINd == False:
        
        for index_row,image in enumerate(temp_df.Image_Name):
            temp_df.loc[index_row,"Hash"] = hasher(image,hash_size=hash_size,bs=True) 
    
    hash_df = temp_df
    #display(hash_df)
   
    zip_ = zip(hash_df.Image_Group,hash_df.Hash)
    
    ## compute many-to-many hamming distances 
    perf_df = pd.DataFrame(columns = ["Hasher","Group_1","Group_2","abs_hash_diff","perc_hash_diff","actual_similarity","predicted_similarity","cf_results"])
    
    counter = 0
    for x,y in itertools.combinations(zip_ ,2):
        perf_df.loc[counter,:]=[np.nan,x[0],y[0],abs(x[1]-y[1]),abs(x[1]-y[1])/(hash_size**2),np.nan,np.nan,np.nan]
        counter+=1

    perf_df.actual_similarity = perf_df.Group_1 == perf_df.Group_2
    perf_df.predicted_similarity = perf_df.perc_hash_diff <= th
    perf_df.Hasher = name
    
    ## compute confusion matrix dimensions 
    for i,t in enumerate(zip(perf_df.actual_similarity,perf_df.predicted_similarity)):
        if t[0] == True and t[1] == True: 
            perf_df.loc[i,"cf_results"] = "TP"
        if t[0] == False and t[1] == False: 
            perf_df.loc[i,"cf_results"] = "TN"
        if t[0] == True and t[1] == False: 
            perf_df.loc[i,"cf_results"] = "FN"
        if t[0] == False and t[1] == True: 
            perf_df.loc[i,"cf_results"] = "FP"
        else: pass
    
    ## calculate performance metrics 
    acc = len(perf_df.loc[(perf_df.cf_results == "TP") | (perf_df.cf_results == "TN")])/perf_df.shape[0]
    recall = len(perf_df.loc[(perf_df.cf_results == "TP")])/len(perf_df.loc[(perf_df.cf_results == "TP") | (perf_df.cf_results == "FN")])
    prec = len(perf_df.loc[(perf_df.cf_results == "TP")]) /len(perf_df.loc[(perf_df.cf_results == "TP") | (perf_df.cf_results == "FP")])
    f1 = 2 * (recall * prec)/(recall + prec)
    
    metrics = [acc,recall,prec,f1]
    metrics_names = ["Accuracy","Recall","Precision","F1_Score"]
    
    metrics_dict = {}
    metrics_dict["Hasher"] = name
    for key,value in zip(metrics_names,metrics):
        metrics_dict[key] = round(value,2)
        
    cfm_results = pd.DataFrame(perf_df.cf_results.value_counts(normalize=True)).T
    cfm_results.index = [name]
    
    if res_print == True:
        print(f"Accuracy: {acc}")
        print(f"Recall: {recall}")
        print(f"Precision: {prec}")
        print(f"F1_Score: {f1}")
        display(perf_df)
    
    return perf_df,metrics_dict,cfm_results
